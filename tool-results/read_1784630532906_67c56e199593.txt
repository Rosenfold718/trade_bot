    50‚Üí    50‚Üí    if (change > 0) gains += change;
    51‚Üí    51‚Üí    else losses += Math.abs(change);

    54‚Üí    54‚Üí  const rs = (gains / period) / (losses / period);

   129‚Üí   129‚Üífunction calcADX(candles: CandleData[], period: number = 14): { adx: number; plusDI: number; minusDI: number } {
   130‚Üí   130‚Üí  if (candles.length < period * 2) return { adx: 0, plusDI: 0, minusDI: 0 };
   131‚Üí   131‚Üí
   132‚Üí   132‚Üí  const trueRanges: number[] = [];
   133‚Üí   133‚Üí  const plusDM: number[] = [];
   134‚Üí   134‚Üí  const minusDM: number[] = [];

   136‚Üí   136‚Üí  for (let i = 1; i < candles.length; i++) {
   137‚Üí   137‚Üí    const high = candles[i].high;
   138‚Üí   138‚Üí    const low = candles[i].low;
   139‚Üí   139‚Üí    const prevHigh = candles[i - 1].high;
   140‚Üí   140‚Üí    const prevLow = candles[i - 1].low;
   141‚Üí   141‚Üí    const prevClose = candles[i - 1].close;
   142‚Üí   142‚Üí
   143‚Üí   143‚Üí    const tr = Math.max(high - low, Math.abs(high - prevClose), Math.abs(low - prevClose));
   144‚Üí   144‚Üí    trueRanges.push(tr);
   145‚Üí   145‚Üí
   146‚Üí   146‚Üí    const upMove = high - prevHigh;
   147‚Üí   147‚Üí    const downMove = prevLow - low;
   148‚Üí   148‚Üí
   149‚Üí   149‚Üí    plusDM.push(upMove > downMove && upMove > 0 ? upMove : 0);
   150‚Üí   150‚Üí    minusDM.push(downMove > upMove && downMove > 0 ? downMove : 0);
   151‚Üí   151‚Üí  }
   152‚Üí   152‚Üí
   153‚Üí   153‚Üí  // Smooth with Wilder's method
   154‚Üí   154‚Üí  const smooth = (data: number[], p: number) => {
   155‚Üí   155‚Üí    const result: number[] = [];
   156‚Üí   156‚Üí    let sum = 0;
   157‚Üí   157‚Üí    for (let i = 0; i < p && i < data.length; i++) sum += data[i];
   158‚Üí   158‚Üí    result.push(sum);
   159‚Üí   159‚Üí    for (let i = p; i < data.length; i++) {
   160‚Üí   160‚Üí      sum = sum - sum / p + data[i];
   161‚Üí   161‚Üí      result.push(sum);
   162‚Üí   162‚Üí    }
   163‚Üí   163‚Üí    return result;
   164‚Üí   164‚Üí  };
   165‚Üí   165‚Üí
   166‚Üí   166‚Üí  const smoothTR = smooth(trueRanges, period);
   167‚Üí   167‚Üí  const smoothPlusDM = smooth(plusDM, period);
   168‚Üí   168‚Üí  const smoothMinusDM = smooth(minusDM, period);
   169‚Üí   169‚Üí
   170‚Üí   170‚Üí  const diValues: number[] = [];
   171‚Üí   171‚Üí  const plusDIValues: number[] = [];
   172‚Üí   172‚Üí  const minusDIValues: number[] = [];
   173‚Üí   173‚Üí
   174‚Üí   174‚Üí  for (let i = 0; i < smoothTR.length; i++) {
   175‚Üí   175‚Üí    const pdi = smoothTR[i] > 0 ? (smoothPlusDM[i] / smoothTR[i]) * 100 : 0;
   176‚Üí   176‚Üí    const mdi = smoothTR[i] > 0 ? (smoothMinusDM[i] / smoothTR[i]) * 100 : 0;
   177‚Üí   177‚Üí    plusDIValues.push(pdi);
   178‚Üí   178‚Üí    minusDIValues.push(mdi);
   179‚Üí   179‚Üí    const diSum = pdi + mdi;
   180‚Üí   180‚Üí    diValues.push(diSum > 0 ? (Math.abs(pdi - mdi) / diSum) * 100 : 0);
   181‚Üí   181‚Üí  }
   182‚Üí   182‚Üí
   183‚Üí   183‚Üí  // Smooth ADX
   184‚Üí   184‚Üí  const adxSmoothed: number[] = [];
   185‚Üí   185‚Üí  if (diValues.length >= period) {
   186‚Üí   186‚Üí    let adxSum = 0;
   187‚Üí   187‚Üí    for (let i = 0; i < period; i++) adxSum += diValues[i];
   188‚Üí   188‚Üí    adxSmoothed.push(adxSum / period);
   189‚Üí   189‚Üí    for (let i = period; i < diValues.length; i++) {
   190‚Üí   190‚Üí      adxSmoothed.push((adxSmoothed[adxSmoothed.length - 1] * (period - 1) + diValues[i]) / period);
   191‚Üí   191‚Üí    }
   192‚Üí   192‚Üí  }
   193‚Üí   193‚Üí
   194‚Üí   194‚Üí  const lastIdx = adxSmoothed.length - 1;
   195‚Üí   195‚Üí  return {
   196‚Üí   196‚Üí    adx: adxSmoothed.length > 0 ? adxSmoothed[lastIdx] : 0,
   197‚Üí   197‚Üí    plusDI: plusDIValues.length > 0 ? plusDIValues[plusDIValues.length - 1] : 0,
   198‚Üí   198‚Üí    minusDI: minusDIValues.length > 0 ? minusDIValues[minusDIValues.length - 1] : 0,
   199‚Üí   199‚Üí  };
   200‚Üí   200‚Üí}

   202‚Üí   202‚Üífunction calcOBV(candles: CandleData[]): { obv: number; trend: number } {
   203‚Üí   203‚Üí  if (candles.length < 2) return { obv: 0, trend: 0 };
   204‚Üí   204‚Üí  let obv = 0;
   205‚Üí   205‚Üí  const obvHistory: number[] = [];
   206‚Üí   206‚Üí  for (let i = 1; i < candles.length; i++) {
   207‚Üí   207‚Üí    if (candles[i].close > candles[i - 1].close) obv += candles[i].volume;
   208‚Üí   208‚Üí    else if (candles[i].close < candles[i - 1].close) obv -= candles[i].volume;
   209‚Üí   209‚Üí    obvHistory.push(obv);

   211‚Üí   211‚Üí  // Simple trend: compare recent OBV vs earlier OBV

   214‚Üí   214‚Üí  const earlier = obvHistory.slice(-20, -10).reduce((a, b) => a + b, 0) / Math.min(10, obvHistory.length - 10);
   215‚Üí   215‚Üí  const trend = earlier !== 0 ? (recent - earlier) / Math.abs(earlier) : 0;
   216‚Üí   216‚Üí  return { obv, trend: Math.max(-1, Math.min(1, trend)) };
   217‚Üí   217‚Üí}

   224‚Üí   224‚Üí  for (const c of slice) {

   228‚Üí   228‚Üí  }
   229‚Üí   229‚Üí  const vwap = cumVolume > 0 ? cumVolumePrice / cumVolume : 0;
   230‚Üí   230‚Üí  const price = candles[candles.length - 1].close;
   231‚Üí   231‚Üí  const signal = vwap > 0 ? (price - vwap) / vwap : 0;

   293‚Üí   293‚Üí    });
   294‚Üí   294‚Üí  } else {
   295‚Üí   295‚Üí    signals.push({ name: 'EMA_200', signal: 0, strength: 0 });
   296‚Üí   296‚Üí  }
   297‚Üí   297‚Üí
   298‚Üí   298‚Üí  // Bollinger Bands
   299‚Üí   299‚Üí  const bb = calcBollingerBands(closes);
   300‚Üí   300‚Üí  const bbWeight = weights['bollinger'] ?? 1;
   301‚Üí   301‚Üí  if (bb.position < 0.1) {
   302‚Üí   302‚Üí    signals.push({ name: 'Bollinger', signal: 1, strength: (0.1 - bb.position) / 0.1 });
   303‚Üí   303‚Üí  } else if (bb.position > 0.9) {
   304‚Üí   304‚Üí    signals.push({ name: 'Bollinger', signal: -1, strength: (bb.position - 0.9) / 0.1 });
   305‚Üí   305‚Üí  } else {
   306‚Üí   306‚Üí    signals.push({ name: 'Bollinger', signal: 0, strength: 0 });
   307‚Üí   307‚Üí  }
   308‚Üí   308‚Üí
   309‚Üí   309‚Üí  // Volume
   310‚Üí   310‚Üí  const volSignal = calcVolumeSignal(candles);
   311‚Üí   311‚Üí  const volWeight = weights['volume'] ?? 1;
   312‚Üí   312‚Üí  signals.push({ name: 'Volume', signal: volSignal > 0 ? 1 : volSignal < 0 ? -1 : 0, strength: Math.abs(volSignal) });
   313‚Üí   313‚Üí
   314‚Üí   314‚Üí  // StochRSI
   315‚Üí   315‚Üí  const stochRSI = calcStochRSI(closes);
   316‚Üí   316‚Üí  if (stochRSI > 0.8) {
   317‚Üí   317‚Üí    signals.push({ name: 'StochRSI', signal: -1, strength: (stochRSI - 0.8) / 0.2 });
   318‚Üí   318‚Üí  } else if (stochRSI < 0.2) {
   319‚Üí   319‚Üí    signals.push({ name: 'StochRSI', signal: 1, strength: (0.2 - stochRSI) / 0.2 });

   323‚Üí   323‚Üí

   334‚Üí   334‚Üí  } else if (adxResult.adx < 20) {
   335‚Üí   335‚Üí    // Weak/ranging ‚Äî avoid, slight neutral
   336‚Üí   336‚Üí    signals.push({ name: 'ADX', signal: 0, strength: 0.1 });
   337‚Üí   337‚Üí  } else {
   338‚Üí   338‚Üí    signals.push({ name: 'ADX', signal: 0, strength: 0 });

   346‚Üí   346‚Üí  if (obvResult.trend > 0.05 && priceChange > 0) {

   348‚Üí   348‚Üí    signals.push({ name: 'OBV', signal: 1, strength: Math.min(Math.abs(obvResult.trend) * 5, 1) });
   349‚Üí   349‚Üí  } else if (obvResult.trend < -0.05 && priceChange < 0) {
   350‚Üí   350‚Üí    // Falling OBV + falling price = bearish confirmation
   351‚Üí   351‚Üí    signals.push({ name: 'OBV', signal: -1, strength: Math.min(Math.abs(obvResult.trend) * 5, 1) });
   352‚Üí   352‚Üí  } else if (obvResult.trend > 0.05 && priceChange < 0) {
   353‚Üí   353‚Üí    // Divergence: OBV rising but price falling ‚Äî potential reversal up
   354‚Üí   354‚Üí    signals.push({ name: 'OBV', signal: 1, strength: Math.min(Math.abs(obvResult.trend) * 3, 0.7) });
   355‚Üí   355‚Üí  } else if (obvResult.trend < -0.05 && priceChange > 0) {
   356‚Üí   356‚Üí    // Divergence: OBV falling but price rising ‚Äî potential reversal down
   357‚Üí   357‚Üí    signals.push({ name: 'OBV', signal: -1, strength: Math.min(Math.abs(obvResult.trend) * 3, 0.7) });
   358‚Üí   358‚Üí  } else {

   361‚Üí   361‚Üí

   366‚Üí   366‚Üí  } else if (vwapResult.signal < -0.005) {
   367‚Üí   367‚Üí    signals.push({ name: 'VWAP', signal: -1, strength: Math.min(Math.abs(vwapResult.signal) * 10, 1) });
   368‚Üí   368‚Üí  } else {
   369‚Üí   369‚Üí    signals.push({ name: 'VWAP', signal: 0, strength: 0 });
   370‚Üí   370‚Üí  }

   379‚Üí   379‚Üíexport function makeTradingDecision(
   380‚Üí   380‚Üí  symbol: string,
   381‚Üí   381‚Üí  candles: CandleData[],
   382‚Üí   382‚Üí  weights: Record<string, number>,
   383‚Üí   383‚Üí  idleMinutes: number = 0,
   384‚Üí   384‚Üí): TradingDecision {
   385‚Üí   385‚Üí  const indicators = analyzeIndicators(candles, weights);
   386‚Üí   386‚Üí  const closes = candles.map(c => c.close);
   387‚Üí   387‚Üí  const price = closes[closes.length - 1];
   388‚Üí   388‚Üí  const atr = calcATR(candles);
   389‚Üí   389‚Üí
   390‚Üí   390‚Üí  // ============================================================
   391‚Üí   391‚Üí  // POINT 1: ADX regime filter ‚Äî skip if market is ranging (ADX < 20)
   392‚Üí   392‚Üí  // ============================================================
   393‚Üí   393‚Üí  const adxResult = calcADX(candles);
   394‚Üí   394‚Üí  if (adxResult.adx < 20) {
   395‚Üí   395‚Üí    // Market is choppy/ranging ‚Äî don't trade
   396‚Üí   396‚Üí    return {
   397‚Üí   397‚Üí      symbol,
   398‚Üí   398‚Üí      direction: 'none',
   399‚Üí   399‚Üí      score: 0,
   400‚Üí   400‚Üí      leverage: 1,
   401‚Üí   401‚Üí      stopLoss: 0,
   402‚Üí   402‚Üí      takeProfit: 0,
   403‚Üí   403‚Üí      indicators,
   404‚Üí   404‚Üí    };
   405‚Üí   405‚Üí  }
   406‚Üí   406‚Üí
   407‚Üí   407‚Üí  let longScore = 0;
   408‚Üí   408‚Üí  let shortScore = 0;
   409‚Üí   409‚Üí  let longCount = 0;  // POINT 3: count agreeing indicators
   410‚Üí   410‚Üí  let shortCount = 0;
   411‚Üí   411‚Üí
   412‚Üí   412‚Üí  for (const ind of indicators) {
   413‚Üí   413‚Üí    const w = weights[ind.name] ?? 1;
   414‚Üí   414‚Üí    if (ind.signal > 0) {
   415‚Üí   415‚Üí      longScore += ind.strength * w;
   416‚Üí   416‚Üí      longCount++;
   417‚Üí   417‚Üí    } else if (ind.signal < 0) {
   418‚Üí   418‚Üí      shortScore += ind.strength * w;
   419‚Üí   419‚Üí      shortCount++;
   420‚Üí   420‚Üí    }
   421‚Üí   421‚Üí  }
   422‚Üí   422‚Üí
   423‚Üí   423‚Üí  const OPEN_THRESHOLD = 0.15;
   424‚Üí   424‚Üí  const absLongScore = Math.abs(longScore);
   425‚Üí   425‚Üí  const absShortScore = Math.abs(shortScore);
   426‚Üí   426‚Üí  const maxScore = Math.max(absLongScore, absShortScore);
   427‚Üí   427‚Üí
   428‚Üí   428‚Üí  // ============================================================
   429‚Üí   429‚Üí  // POINT 3: Confluence filter ‚Äî require ‚â•5 indicators to agree
   430‚Üí   430‚Üí  // ============================================================
   431‚Üí   431‚Üí  const bestCount = Math.max(longCount, shortCount);
   432‚Üí   432‚Üí  if (bestCount < 5) {
   433‚Üí   433‚Üí    return {
   434‚Üí   434‚Üí      symbol,
   435‚Üí   435‚Üí      direction: 'none',
   436‚Üí   436‚Üí      score: maxScore,
   437‚Üí   437‚Üí      leverage: 1,
   438‚Üí   438‚Üí      stopLoss: 0,
   439‚Üí   439‚Üí      takeProfit: 0,
   440‚Üí   440‚Üí      indicators,
   441‚Üí   441‚Üí    };
   442‚Üí   442‚Üí  }
   443‚Üí   443‚Üí
   444‚Üí   444‚Üí  // ============================================================
   445‚Üí   445‚Üí  // POINT 2: Removed sub-threshold fallback ‚Äî only trade with real signals
   446‚Üí   446‚Üí  // ============================================================
   447‚Üí   447‚Üí  let direction: 'long' | 'short' | 'none' = 'none';
   448‚Üí   448‚Üí  let score = 0;
   449‚Üí   449‚Üí
   450‚Üí   450‚Üí  if (absLongScore >= OPEN_THRESHOLD && absLongScore >= absShortScore) {
   451‚Üí   451‚Üí    direction = 'long';
   452‚Üí   452‚Üí    score = longScore;
   453‚Üí   453‚Üí  } else if (absShortScore >= OPEN_THRESHOLD && absShortScore > absLongScore) {
   454‚Üí   454‚Üí    direction = 'short';
   455‚Üí   455‚Üí    score = shortScore;
   456‚Üí   456‚Üí  }
   457‚Üí   457‚Üí  // No more fallback at score > 0.02
   458‚Üí   458‚Üí
   459‚Üí   459‚Üí  // Leverage based on signal strength (1x to 10x), lower for weak signals
   460‚Üí   460‚Üí  const leverage = direction === 'none' ? 1 : Math.min(10, Math.max(1, Math.round(maxScore * 3)));
   461‚Üí   461‚Üí
   462‚Üí   462‚Üí  // Stop loss and take profit based on ATR
   463‚Üí   463‚Üí  const stopLossPercent = atr / price;
   464‚Üí   464‚Üí  const takeProfitPercent = stopLossPercent * 2.5; // Improved: 1:2.5 risk/reward
   465‚Üí   465‚Üí
   466‚Üí   466‚Üí  const stopLoss = direction === 'long'
   467‚Üí   467‚Üí    ? price * (1 - stopLossPercent)
   468‚Üí   468‚Üí    : price * (1 + stopLossPercent);
   469‚Üí   469‚Üí  const takeProfit = direction === 'long'
   470‚Üí   470‚Üí    ? price * (1 + takeProfitPercent)
   471‚Üí   471‚Üí    : price * (1 - takeProfitPercent);
   472‚Üí   472‚Üí
   473‚Üí   473‚Üí  return {
   474‚Üí   474‚Üí    symbol,
   475‚Üí   475‚Üí    direction,
   476‚Üí   476‚Üí    score,
   477‚Üí   477‚Üí    leverage,
   478‚Üí   478‚Üí    stopLoss,
   479‚Üí   479‚Üí    takeProfit,
   480‚Üí   480‚Üí    indicators,
   481‚Üí   481‚Üí  };
   482‚Üí   482‚Üí}
   483‚Üí   483‚Üí
   484‚Üí   484‚Üí// ============================================================
   485‚Üí   485‚Üí// Backtesting
   486‚Üí   486‚Üí// ============================================================
   487‚Üí   487‚Üí
   488‚Üí   488‚Üíexport function runBacktest(
   489‚Üí   489‚Üí  symbol: string,
   490‚Üí   490‚Üí  candles: CandleData[],
   491‚Üí   491‚Üí  weights: Record<string, number>,
   492‚Üí   492‚Üí  initialBalance: number = 100,
   493‚Üí   493‚Üí): BacktestSummary {
   494‚Üí   494‚Üí  const minCandles = 200;
   495‚Üí   495‚Üí  if (candles.length < minCandles) {
   496‚Üí   496‚Üí    return {
   497‚Üí   497‚Üí      symbol,
   498‚Üí   498‚Üí      total_trades: 0,
   499‚Üí   499‚Üí      winning_trades: 0,
   500‚Üí   500‚Üí      losing_trades: 0,
   501‚Üí   501‚Üí      total_pnl: 0,
   502‚Üí   502‚Üí      winrate: 0,
   503‚Üí   503‚Üí      profit_factor: 0,
   504‚Üí   504‚Üí      indicator_performance: {},
   505‚Üí   505‚Üí    };
   506‚Üí   506‚Üí  }
   507‚Üí   507‚Üí
   508‚Üí   508‚Üí  let balance = initialBalance;
   509‚Üí   509‚Üí  const trades: BacktestTrade[] = [];
   510‚Üí   510‚Üí  const indicatorPerformance: Record<string, { wins: number; losses: number; pnl: number }> = {};
   511‚Üí   511‚Üí  const indicatorNames = ['RSI', 'MACD', 'EMA_50', 'EMA_200', 'Bollinger', 'Volume', 'StochRSI', 'ADX', 'OBV', 'VWAP'];
   512‚Üí   512‚Üí  for (const name of indicatorNames) {
   513‚Üí   513‚Üí    indicatorPerformance[name] = { wins: 0, losses: 0, pnl: 0 };
   514‚Üí   514‚Üí  }
   515‚Üí   515‚Üí
   516‚Üí   516‚Üí  let i = minCandles;
   517‚Üí   517‚Üí  while (i < candles.length - 1) {
   518‚Üí   518‚Üí    const historicalCandles = candles.slice(0, i + 1);
   519‚Üí   519‚Üí    const decision = makeTradingDecision(symbol, historicalCandles, weights);
   520‚Üí   520‚Üí
   521‚Üí   521‚Üí    if (decision.direction !== 'none' && balance > 0) {
   522‚Üí   522‚Üí      const amount = balance * 0.1 * decision.leverage;
   523‚Üí   523‚Üí      if (amount < 0.01) { i++; continue; }
   524‚Üí   524‚Üí
   525‚Üí   525‚Üí      const entryPrice = candles[i].close;
   526‚Üí   526‚Üí      let exitPrice = entryPrice;
   527‚Üí   527‚Üí      let j = i + 1;
   528‚Üí   528‚Üí      const maxHold = Math.min(i + 24, candles.length); // max 24h hold
   529‚Üí   529‚Üí
   530‚Üí   530‚Üí      while (j < maxHold) {
   531‚Üí   531‚Üí        const candle = candles[j];
   532‚Üí   532‚Üí        if (decision.direction === 'long') {
   533‚Üí   533‚Üí          if (candle.low <= decision.stopLoss) { exitPrice = decision.stopLoss; break; }
   534‚Üí   534‚Üí          if (candle.high >= decision.takeProfit) { exitPrice = decision.takeProfit; break; }
   535‚Üí   535‚Üí        } else {
   536‚Üí   536‚Üí          if (candle.high >= decision.stopLoss) { exitPrice = decision.stopLoss; break; }
   537‚Üí   537‚Üí          if (candle.low <= decision.takeProfit) { exitPrice = decision.takeProfit; break; }
   538‚Üí   538‚Üí        }
   539‚Üí   539‚Üí        exitPrice = candle.close;
   540‚Üí   540‚Üí        j++;
   541‚Üí   541‚Üí      }
   542‚Üí   542‚Üí
   543‚Üí   543‚Üí      const priceChange = decision.direction === 'long'
   544‚Üí   544‚Üí        ? (exitPrice - entryPrice) / entryPrice
   545‚Üí   545‚Üí        : (entryPrice - exitPrice) / entryPrice;
   546‚Üí   546‚Üí      const pnl = (amount * priceChange) - (amount * 0.001); // 0.1% fee
   547‚Üí   547‚Üí      balance += pnl;
   548‚Üí   548‚Üí
   549‚Üí   549‚Üí      const activeIndicators = decision.indicators
   550‚Üí   550‚Üí        .filter(ind => ind.signal !== 0)
   551‚Üí   551‚Üí        .map(ind => ind.name);
   552‚Üí   552‚Üí
   553‚Üí   553‚Üí      const isWin = pnl > 0;
   554‚Üí   554‚Üí      trades.push({
   555‚Üí   555‚Üí        symbol,
   556‚Üí   556‚Üí        entry_price: entryPrice,
   557‚Üí   557‚Üí        exit_price: exitPrice,
   558‚Üí   558‚Üí        amount,
   559‚Üí   559‚Üí        leverage: decision.leverage,
   560‚Üí   560‚Üí        direction: decision.direction,
   561‚Üí   561‚Üí        pnl,
   562‚Üí   562‚Üí        indicators_used: activeIndicators,
   563‚Üí   563‚Üí      });
   564‚Üí   564‚Üí
   565‚Üí   565‚Üí      for (const indName of activeIndicators) {
   566‚Üí   566‚Üí        if (indicatorPerformance[indName]) {
   567‚Üí   567‚Üí          if (isWin) {
   568‚Üí   568‚Üí            indicatorPerformance[indName].wins++;
   569‚Üí   569‚Üí            indicatorPerformance[indName].pnl += pnl;
   570‚Üí   570‚Üí          } else {
   571‚Üí   571‚Üí            indicatorPerformance[indName].losses++;
   572‚Üí   572‚Üí            indicatorPerformance[indName].pnl += pnl;
   573‚Üí   573‚Üí          }
   574‚Üí   574‚Üí        }
   575‚Üí   575‚Üí      }
   576‚Üí   576‚Üí
   577‚Üí   577‚Üí      i = j; // Move past the trade duration
   578‚Üí   578‚Üí    } else {
   579‚Üí   579‚Üí      i++;
   580‚Üí   580‚Üí    }
   581‚Üí   581‚Üí  }
   582‚Üí   582‚Üí
   583‚Üí   583‚Üí  const winningTrades = trades.filter(t => t.pnl > 0);
   584‚Üí   584‚Üí  const losingTrades = trades.filter(t => t.pnl <= 0);
   585‚Üí   585‚Üí  const totalWins = winningTrades.reduce((s, t) => s + t.pnl, 0);
   586‚Üí   586‚Üí  const totalLosses = Math.abs(losingTrades.reduce((s, t) => s + t.pnl, 0));
   587‚Üí   587‚Üí  const winrate = trades.length > 0 ? (winningTrades.length / trades.length) * 100 : 0;
   588‚Üí   588‚Üí  const profitFactor = totalLosses > 0 ? totalWins / totalLosses : totalWins > 0 ? 999 : 0;
   589‚Üí   589‚Üí
   590‚Üí   590‚Üí  return {
   591‚Üí   591‚Üí    symbol,
   592‚Üí   592‚Üí    total_trades: trades.length,
   593‚Üí   593‚Üí    winning_trades: winningTrades.length,
   594‚Üí   594‚Üí    losing_trades: losingTrades.length,
   595‚Üí   595‚Üí    total_pnl: balance - initialBalance,
   596‚Üí   596‚Üí    winrate,
   597‚Üí   597‚Üí    profit_factor,
   598‚Üí   598‚Üí    indicator_performance: indicatorPerformance,
   599‚Üí   599‚Üí  };
   600‚Üí   600‚Üí}
   601‚Üí   601‚Üí
   602‚Üí   602‚Üí// ============================================================
   603‚Üí   603‚Üí// Order Book Analysis
   604‚Üí   604‚Üí// ============================================================
   605‚Üí   605‚Üí
   606‚Üí   606‚Üíexport function analyzeOrderBook(
   607‚Üí   607‚Üí  bids: Array<{ price: number; quantity: number; total: number }>,
   608‚Üí   608‚Üí  asks: Array<{ price: number; quantity: number; total: number }>,
   609‚Üí   609‚Üí  midPrice: number
   610‚Üí   610‚Üí): IndicatorSignal {
   611‚Üí   611‚Üí  // Calculate bid/ask volume imbalance
   612‚Üí   612‚Üí  const totalBidVol = bids.reduce((sum, b) => sum + b.quantity, 0);
   613‚Üí   613‚Üí  const totalAskVol = asks.reduce((sum, a) => sum + a.quantity, 0);
   614‚Üí   614‚Üí  const totalVol = totalBidVol + totalAskVol;
   615‚Üí   615‚Üí
   616‚Üí   616‚Üí  let imbalance = 0;
   617‚Üí   617‚Üí  if (totalVol > 0) {
   618‚Üí   618‚Üí    imbalance = (totalBidVol - totalAskVol) / totalVol;
   619‚Üí   619‚Üí  }
   620‚Üí   620‚Üí
   621‚Üí   621‚Üí  // Detect walls ‚Äî large orders (>3x average) near price
   622‚Üí   622‚Üí  let bidWallPressure = 0;
   623‚Üí   623‚Üí  let askWallPressure = 0;
   624‚Üí   624‚Üí
   625‚Üí   625‚Üí  if (bids.length > 1) {
   626‚Üí   626‚Üí    const avgBidQty = totalBidVol / bids.length;
   627‚Üí   627‚Üí    const largeBids = bids.filter(b => b.quantity > avgBidQty * 3);
   628‚Üí   628‚Üí    bidWallPressure = largeBids.reduce((sum, b) => {
   629‚Üí   629‚Üí      const distance = (midPrice - b.price) / midPrice;
   630‚Üí   630‚Üí      return sum + (b.quantity * Math.exp(-distance * 100)); // closer walls matter more
   631‚Üí   631‚Üí    }, 0);
   632‚Üí   632‚Üí  }
   633‚Üí   633‚Üí
   634‚Üí   634‚Üí  if (asks.length > 1) {
   635‚Üí   635‚Üí    const avgAskQty = totalAskVol / asks.length;
   636‚Üí   636‚Üí    const largeAsks = asks.filter(a => a.quantity > avgAskQty * 3);
   637‚Üí   637‚Üí    askWallPressure = largeAsks.reduce((sum, a) => {
   638‚Üí   638‚Üí      const distance = (a.price - midPrice) / midPrice;
   639‚Üí   639‚Üí      return sum + (a.quantity * Math.exp(-distance * 100));
   640‚Üí   640‚Üí    }, 0);
   641‚Üí   641‚Üí  }
   642‚Üí   642‚Üí
   643‚Üí   643‚Üí  // Combine signals
   644‚Üí   644‚Üí  const wallImbalance = bidWallPressure > 0 || askWallPressure > 0
   645‚Üí   645‚Üí    ? (bidWallPressure - askWallPressure) / (bidWallPressure + askWallPressure)
   646‚Üí   646‚Üí    : 0;
   647‚Üí   647‚Üí
   648‚Üí   648‚Üí  const combinedSignal = imbalance * 0.6 + wallImbalance * 0.4;
   649‚Üí   649‚Üí
   650‚Üí   650‚Üí  let signal: number;
   651‚Üí   651‚Üí  let strength: number;
   652‚Üí   652‚Üí
   653‚Üí   653‚Üí  if (combinedSignal > 0.15) {
   654‚Üí   654‚Üí    signal = 1;
   655‚Üí   655‚Üí    strength = Math.min(combinedSignal / 0.5, 1);
   656‚Üí   656‚Üí  } else if (combinedSignal < -0.15) {
   657‚Üí   657‚Üí    signal = -1;
   658‚Üí   658‚Üí    strength = Math.min(Math.abs(combinedSignal) / 0.5, 1);
   659‚Üí   659‚Üí  } else {
   660‚Üí   660‚Üí    signal = 0;
   661‚Üí   661‚Üí    strength = 0;
   662‚Üí   662‚Üí  }
   663‚Üí   663‚Üí
   664‚Üí   664‚Üí  return {
   665‚Üí   665‚Üí    name: 'OrderBook',
   666‚Üí   666‚Üí    signal,
   667‚Üí   667‚Üí    strength,
   668‚Üí   668‚Üí  };
   669‚Üí   669‚Üí}
   670‚Üí   670‚Üí
   671‚Üí   671‚Üí// ============================================================
   672‚Üí   672‚Üí// Weight Optimization (Post-Backtest Learning)
   673‚Üí   673‚Üí// ============================================================
   674‚Üí   674‚Üí
   675‚Üí   675‚Üíexport function optimizeWeights(
   676‚Üí   676‚Üí  currentWeights: Record<string, number>,
   677‚Üí   677‚Üí  allSummaries: BacktestSummary[],
   678‚Üí   678‚Üí): Record<string, number> {
   679‚Üí   679‚Üí  const newWeights: Record<string, number> = { ...currentWeights };
   680‚Üí   680‚Üí  const indicatorAgg: Record<string, { totalWins: number; totalTrades: number; totalPnl: number }> = {};
   681‚Üí   681‚Üí
   682‚Üí   682‚Üí  for (const summary of allSummaries) {
   683‚Üí   683‚Üí    for (const [name, perf] of Object.entries(summary.indicator_performance)) {
   684‚Üí   684‚Üí      if (!indicatorAgg[name]) indicatorAgg[name] = { totalWins: 0, totalTrades: 0, totalPnl: 0 };
   685‚Üí   685‚Üí      indicatorAgg[name].totalWins += perf.wins;
   686‚Üí   686‚Üí      indicatorAgg[name].totalTrades += perf.wins + perf.losses;
   687‚Üí   687‚Üí      indicatorAgg[name].totalPnl += perf.pnl;
   688‚Üí   688‚Üí    }
   689‚Üí   689‚Üí  }
   690‚Üí   690‚Üí
   691‚Üí   691‚Üí  for (const [name, agg] of Object.entries(indicatorAgg)) {
   692‚Üí   692‚Üí    if (agg.totalTrades < 3) continue; // Not enough data
   693‚Üí   693‚Üí    const indicatorWinrate = agg.totalWins / agg.totalTrades;
   694‚Üí   694‚Üí    // Scale weight: bad indicators get 0.2, good ones get up to 2.5
   695‚Üí   695‚Üí    if (indicatorWinrate > 0.55) {
   696‚Üí   696‚Üí      newWeights[name] = Math.min(2.5, 1.0 + (indicatorWinrate - 0.5) * 5);
   697‚Üí   697‚Üí    } else if (indicatorWinrate < 0.45) {
   698‚Üí   698‚Üí      newWeights[name] = Math.max(0.2, 1.0 - (0.5 - indicatorWinrate) * 5);
   699‚Üí   699‚Üí    } else {
   700‚Üí   700‚Üí      newWeights[name] = 1.0;
   701‚Üí   701‚Üí    }
   702‚Üí   702‚Üí  }
   703‚Üí   703‚Üí
   704‚Üí   704‚Üí  return newWeights;
   705‚Üí   705‚Üí}
